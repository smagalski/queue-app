36/// The options for the open dialog API.
37#[derive(Debug, Clone, Deserialize)]
38#[serde(rename_all = "camelCase")]
39pub struct OpenDialogOptions {
40    /// The title of the dialog window.
41    title: Option<String>,
42    /// The filters of the dialog.
43    #[serde(default)]
44    filters: Vec<DialogFilter>,
45    /// Whether the dialog allows multiple selection or not.
46    #[serde(default)]
47    multiple: bool,
48    /// Whether the dialog is a directory selection (`true` value) or file selection (`false` value).
49    #[serde(default)]
50    directory: bool,
51    /// The initial path of the dialog.
52    default_path: Option<PathBuf>,
53    /// If [`Self::directory`] is true, indicates that it will be read recursively later.
54    /// Defines whether subdirectories will be allowed on the scope or not.
55    #[serde(default)]
56    #[cfg_attr(mobile, allow(dead_code))]
57    recursive: bool,
58    /// Whether to allow creating directories in the dialog **macOS Only**
59    can_create_directories: Option<bool>,
60    /// The preferred mode of the dialog.
61    /// This is meant for mobile platforms (iOS and Android) which have distinct file and media pickers.
62    /// On desktop, this option is ignored.
63    /// If not provided, the dialog will automatically choose the best mode based on the MIME types of the filters.
64    #[serde(default)]
65    #[cfg_attr(mobile, allow(dead_code))]
66    picker_mode: Option<PickerMode>,
67    /// The file access mode of the dialog.
68    #[serde(default)]
69    #[cfg_attr(mobile, allow(dead_code))]
70    file_access_mode: Option<FileAccessMode>,
71}

89#[cfg(mobile)]
90fn set_default_path<R: Runtime>(
91    mut dialog_builder: FileDialogBuilder<R>,
92    default_path: PathBuf,
93) -> FileDialogBuilder<R> {
94    if let Some(file_name) = default_path.file_name() {
95        dialog_builder = dialog_builder.set_file_name(file_name.to_string_lossy());
96    }
97    dialog_builder
98}
99
100#[cfg(desktop)]
101fn set_default_path<R: Runtime>(
102    mut dialog_builder: FileDialogBuilder<R>,
103    default_path: PathBuf,
104) -> FileDialogBuilder<R> {
105    // we need to adjust the separator on Windows: https://github.com/tauri-apps/tauri/issues/8074
106    let default_path: PathBuf = default_path.components().collect();
107    if default_path.is_file() || !default_path.exists() {
108        if let (Some(parent), Some(file_name)) = (default_path.parent(), default_path.file_name()) {
109            if parent.components().count() > 0 {
110                dialog_builder = dialog_builder.set_directory(parent);
111            }
112            dialog_builder = dialog_builder.set_file_name(file_name.to_string_lossy());
113        } else {
114            dialog_builder = dialog_builder.set_directory(default_path);
115        }
116        dialog_builder
117    } else {
118        dialog_builder.set_directory(default_path)
119    }
120}

122#[command]
123pub(crate) async fn open<R: Runtime>(
124    window: Window<R>,
125    dialog: State<'_, Dialog<R>>,
126    options: OpenDialogOptions,
127) -> Result<OpenResponse> {
128    let mut dialog_builder = dialog.file();
129    #[cfg(any(windows, target_os = "macos"))]
130    {
131        dialog_builder = dialog_builder.set_parent(&window);
132    }
133    if let Some(title) = options.title {
134        dialog_builder = dialog_builder.set_title(title);
135    }
136    if let Some(default_path) = options.default_path {
137        dialog_builder = set_default_path(dialog_builder, default_path);
138    }
139    if let Some(can) = options.can_create_directories {
140        dialog_builder = dialog_builder.set_can_create_directories(can);
141    }
142    if let Some(picker_mode) = options.picker_mode {
143        dialog_builder = dialog_builder.set_picker_mode(picker_mode);
144    }
145    for filter in options.filters {
146        let extensions: Vec<&str> = filter.extensions.iter().map(|s| &**s).collect();
147        dialog_builder = dialog_builder.add_filter(filter.name, &extensions);
148    }
149    if let Some(file_access_mode) = options.file_access_mode {
150        dialog_builder = dialog_builder.set_file_access_mode(file_access_mode);
151    }
152
153    let res = if options.directory {
154        #[cfg(desktop)]
155        {
156            let tauri_scope = window.state::<tauri::scope::Scopes>();
157
158            if options.multiple {
159                let folders = dialog_builder.blocking_pick_folders();
160                if let Some(folders) = &folders {
161                    for folder in folders {
162                        if let Ok(path) = folder.clone().into_path() {
163                            if let Some(s) = window.try_fs_scope() {
164                                s.allow_directory(&path, options.recursive)?;
165                            }
166                            tauri_scope.allow_directory(&path, options.directory)?;
167                        }
168                    }
169                }
170                OpenResponse::Folders(
171                    folders.map(|folders| folders.into_iter().map(|p| p.simplified()).collect()),
172                )
173            } else {
174                let folder = dialog_builder.blocking_pick_folder();
175                if let Some(folder) = &folder {
176                    if let Ok(path) = folder.clone().into_path() {
177                        if let Some(s) = window.try_fs_scope() {
178                            s.allow_directory(&path, options.recursive)?;
179                        }
180                        tauri_scope.allow_directory(&path, options.directory)?;
181                    }
182                }
183                OpenResponse::Folder(folder.map(|p| p.simplified()))
184            }
185        }
186        #[cfg(mobile)]
187        return Err(crate::Error::FolderPickerNotImplemented);
188    } else if options.multiple {
189        let tauri_scope = window.state::<tauri::scope::Scopes>();
190
191        let files = dialog_builder.blocking_pick_files();
192        if let Some(files) = &files {
193            for file in files {
194                if let Ok(path) = file.clone().into_path() {
195                    if let Some(s) = window.try_fs_scope() {
196                        s.allow_file(&path)?;
197                    }
198
199                    tauri_scope.allow_file(&path)?;
200                }
201            }
202        }
203        OpenResponse::Files(files.map(|files| files.into_iter().map(|f| f.simplified()).collect()))
204    } else {
205        let tauri_scope = window.state::<tauri::scope::Scopes>();
206        let file = dialog_builder.blocking_pick_file();
207
208        if let Some(file) = &file {
209            if let Ok(path) = file.clone().into_path() {
210                if let Some(s) = window.try_fs_scope() {
211                    s.allow_file(&path)?;
212                }
213                tauri_scope.allow_file(&path)?;
214            }
215        }
216        OpenResponse::File(file.map(|f| f.simplified()))
217    };
218    Ok(res)
219}

221#[allow(unused_variables)]
222#[command]
223pub(crate) async fn save<R: Runtime>(
224    window: Window<R>,
225    dialog: State<'_, Dialog<R>>,
226    options: SaveDialogOptions,
227) -> Result<Option<FilePath>> {
228    let mut dialog_builder = dialog.file();
229    #[cfg(desktop)]
230    {
231        dialog_builder = dialog_builder.set_parent(&window);
232    }
233    if let Some(title) = options.title {
234        dialog_builder = dialog_builder.set_title(title);
235    }
236    if let Some(default_path) = options.default_path {
237        dialog_builder = set_default_path(dialog_builder, default_path);
238    }
239    if let Some(can) = options.can_create_directories {
240        dialog_builder = dialog_builder.set_can_create_directories(can);
241    }
242    for filter in options.filters {
243        let extensions: Vec<&str> = filter.extensions.iter().map(|s| &**s).collect();
244        dialog_builder = dialog_builder.add_filter(filter.name, &extensions);
245    }
246
247    let tauri_scope = window.state::<tauri::scope::Scopes>();
248
249    let path = dialog_builder.blocking_save_file();
250    if let Some(p) = &path {
251        if let Ok(path) = p.clone().into_path() {
252            if let Some(s) = window.try_fs_scope() {
253                s.allow_file(&path)?;
254            }
255            tauri_scope.allow_file(&path)?;
256        }
257    }
258
259    Ok(path.map(|p| p.simplified()))
260}

262fn message_dialog<R: Runtime>(
263    #[allow(unused_variables)] window: Window<R>,
264    dialog: State<'_, Dialog<R>>,
265    title: Option<String>,
266    message: String,
267    kind: Option<MessageDialogKind>,
268    buttons: MessageDialogButtons,
269) -> MessageDialogBuilder<R> {
270    let mut builder = dialog.message(message);
271
272    builder = builder.buttons(buttons);
273
274    if let Some(title) = title {
275        builder = builder.title(title);
276    }
277
278    #[cfg(desktop)]
279    {
280        builder = builder.parent(&window);
281    }
282
283    if let Some(kind) = kind {
284        builder = builder.kind(kind);
285    }
286
287    builder
288}

290#[command]
291pub(crate) async fn message<R: Runtime>(
292    window: Window<R>,
293    dialog: State<'_, Dialog<R>>,
294    title: Option<String>,
295    message: String,
296    kind: Option<MessageDialogKind>,
297    ok_button_label: Option<String>,
298    buttons: Option<MessageDialogButtons>,
299) -> Result<MessageDialogResult> {
300    let buttons = buttons.unwrap_or(if let Some(ok_button_label) = ok_button_label {
301        MessageDialogButtons::OkCustom(ok_button_label)
302    } else {
303        MessageDialogButtons::Ok
304    });
305
306    Ok(message_dialog(window, dialog, title, message, kind, buttons).blocking_show_with_result())
307}

309#[command]
310pub(crate) async fn ask<R: Runtime>(
311    window: Window<R>,
312    dialog: State<'_, Dialog<R>>,
313    title: Option<String>,
314    message: String,
315    kind: Option<MessageDialogKind>,
316    yes_button_label: Option<String>,
317    no_button_label: Option<String>,
318) -> Result<bool> {
319    let dialog = message_dialog(
320        window,
321        dialog,
322        title,
323        message,
324        kind,
325        if let Some(yes_button_label) = yes_button_label {
326            MessageDialogButtons::OkCancelCustom(
327                yes_button_label,
328                no_button_label.unwrap_or(NO.to_string()),
329            )
330        } else if let Some(no_button_label) = no_button_label {
331            MessageDialogButtons::OkCancelCustom(YES.to_string(), no_button_label)
332        } else {
333            MessageDialogButtons::YesNo
334        },
335    );
336
337    Ok(dialog.blocking_show())
338}
